\def\fibelprintmargin{
	\Ifstr{\currentpagestyle}{scrheadings}{

				\node[rotate=270] at ($(current page.center) + (9.9cm, -10.9cm)$)
				{\fibelmargin};

				\node[rotate=90] at ($(current page.center) + (-9.9cm, -10.9cm)$)
				{\fibelmargin};
			\end{tikzpicture}
		}
	}{}
}
